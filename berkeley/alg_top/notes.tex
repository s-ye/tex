Let $\Gr(k,n)$ denote the Grassmannian of $k$-planes in $V$, a real or complex vector space of dimension $n$. We will not distinguish since the statements we make will be true for both cases. For the real Grassmannian, we take homology with $\Z/2$ coefficients, and for the complex Grassmannian, we take homology with $\Z$ coefficients.
\begin{theorem} 
    For a CW complex $X$, $\R, \C, \H$-vector bundles over $X$ up to isomorphism are classified by homotopy classes of maps to the infinite Grassmannian $Gr_k(\mathbb{C}^\infty) = \lim_{n\to\infty} Gr_k(\mathbb{C}^n)$.
\end{theorem}

\begin{remark}
    $\Sp(n) = U(2n) \cap GL(n, \H)$ is the compact symplectic group acts on $\H^n$ by left multiplication and commutes with the right $\H$-action which is the structure map for $\H$-vector bundles. 
\end{remark}
Recall that the Schubert stratification is given by row echelon form for matrices. Let $T\subset V$ be a subspace of dimension $k$. Write $T$ as the row space of a $k\times n$ matrix. We can perform row operations to put it in row echelon form, and this depends only on $T$. In particular, $T$ has a canonical basis but it does not depend continuously on $T$. It may jump whenever the pivots move. The Schubert stratification is given by the locus where the pivots are fixed. Therefore, it is enough to give an increasing sequence $\sigma_p$ of pivot positions or a sequence $s$ of $0$'s and $1$'s of length $n$ with exactly $k$ $1$'s. 

\begin{example}
    Let us row reduce from right to left. For example in $\Gr(2,4)$ \begin{align*}
        \begin{bmatrix}
            * & 0 & * & 1 \\
            * & 1 & 0 & 0
        \end{bmatrix}
    \end{align*}
    corresponds to the sequence $s = (0,1,0,1)$ and the pivot positions $\sigma_1 = 2, \sigma_2 = 4$. The dimension is given by \begin{align*}
        \dim S_\sigma = \sum_{p=1}^k (\sigma_p - p) = (2-1) + (4-2) = 3 = \sum_{1s \in s} \text{total number of 0s to the left} = 1 + 2
    \end{align*}
\end{example}

\begin{corollary}
    [Approximation theorem] If $\dim X < n-k$ or $\dim X < 2(n-k)$ in the complex case, then a map $X \to \Gr_\R(k,n+1)$ or $X \to \Gr_\C(k,n+1)$ can be homotoped to a map $X \to \Gr(k,n)$. In particular $\Gr(k,n+1)/\Gr(k,n)$ is $(n-k-1)$-connected or $(2(n-k)-1)$-connected in the complex case.

    If $\dim X < k+1$ or $\dim X < 2(k+1)$ in the complex case, then a map $X \to \Gr_\R(k+1,n+1)$ or $X \to \Gr_\C(k+1,n+1)$ can be homotoped to a map $X \to \Gr(k,n)$. In particular $\Gr(k+1,n+1)/\Gr(k,n)$ is $k$-connected or $2k+1$-connected in the complex case.
\end{corollary}


\begin{corollary}
    [Stability for classification of vector bundles] If $\dim X < n-k-1$ or $\dim X < 2(n-k)-1$ in the complex case, then $\Vect^k(X) = [X, \Gr(k,n)]$.
\end{corollary}

\subsection{Stiefel manifodls and classifing spaces}
Recall that $T^k \to \Gr(k,n)$ is the tautological subbundle of $\R^n$ or $\C^n$. We have the frame bundle $\St(k,n)$ given by basis of $k$-planes. Gram Schmidt tells us that orthonormal frames $\hookrightarrow$ frames is a homotopy equivalence so we have $\St(k,n) \to \Gr(k,n)$ is a principal $GL_k$-bundle and $\St^{on}(k,n) \to \Gr(k,n)$ is a principal $O(k)$-bundle or $U(k)$-bundle. 

What is the codimension of the complement of $\St(k,n)$ in $\Mat_{k,n}$? Well $\Mat(k,n)$ is stratified by rank and the largest stratum in the complement of $\St(k,n)$ is given by rank $k-1$ matrices which has codimension $n-k+1$. 

\begin{corollary}
    If $\dim X < n-k$ or $\dim X < 2(n-k)+1$ then $[X,\St(k,n)] = *$. In particiular $\St(k,\infty)$ and $\St^{on}(k,\infty)$ are contractible.
\end{corollary}

Thus we have stumbled upon the fact that $\St^{on}(k,\infty)$ is a contractible space with a free $O(k)$ or $U(k)$ action and $\Gr(k,\infty)$ is the quotient. 

\begin{definition}
    Let $G$ be a topological group which is nice in the sense that it has the homotopy type of a CW complex. A CW complex with a principal $G$ bundle $EG\to BG$ with contractible total space is called a \textbf{classifing space} for $G$.
\end{definition}

\begin{theorem}
    $EG\to BG$ is unique up to homology equivalence, canonical up to homotopy if you choose the base point $*$ in $BG$ and identify the fiber over $*$ with $G$. Moreover, isomorphism classes of principal $G$-bundles over a CW complex $X,*$ with trivialization over $*$ are in bijection with homotopy classes of pointed maps $[X, BG]$.
\end{theorem}
In fact $(2)$ implies $(1)$ because we have identified $BG$ as a functor out of the homotopy category of pointed CW complexes, and the functor is unique up to natural isomorphism.

\begin{remark}
    $G$ is connected if and only if $BG$ is simply connected. Extreme cases are $G$ is discrete, then $EG$ is the universal cover of $BG$. 
\end{remark}

\begin{remark}
    $G$ discrete, $X$ is connected. Then $[X, BG]$ is the set of homomorphisms $\pi_1(X) \to G$ up to conjugation, whereas $[X, BG]_* = \Hom(\pi_1(X), G)$. The first is isomorphism classes of principal $G$-bundles, the second is isomorphism classes of principal $G$-bundles with a trivialization at the base point.
\end{remark}

\begin{proposition}
    The total space of the fiber bundle $\St^{on}(n+1,\infty) \times S^{2n+1} \to \Gr(n+1,\infty)$ is homotopy equivalent to $\Gr(n,\infty)$ and the fiber bundle map is homotopic to the inclusion $\Gr(n,\infty) \to \Gr(n+1,\infty)$. \end{proposition}

\begin{proof}
    A map $X \to \St_\C(n+1,\infty)\times_{U(n+1)} S^{2n+1}$ classifies up to isomorphism a complex vector bundle of rank $n+1$ together with a section of unit norm. This is equivalent (up to isomorphism and rescaling of hermitian metrics (which is a contractible space)) to $E^{n+1} \to X, s$ where $s$ is a nonvanishing section. This is equivalent to a rank $n$ vector bundle by taking the quotient. Thus the target space $\St_\C(n+1,\infty)\times_{U(n+1)} S^{2n+1}$ is a classifying space for rank $n$ vector bundles, and thus homotopy equivalent to $\Gr(n,\infty)$. The map to the base $\Gr(n+1,\infty)$ classifies the stabilized bundle of rank $n+1$ and so is homotopy equivalent to natural map.
\end{proof}

\subsection{Three classical theorems}
\begin{theorem}[Leray Hirsch]
    Let $F\to E \xrightarrow{p} B$ be a connected fiber bundle over a CW base. Let $R$ be a coefficient ring for cohomology, i.e a quotient of $\Z$ or somehting between $\Z$ or $\Q$. Suppose $H^*(F;R)$ is a free $R$-module on $\alpha_1,\dots, \alpha_k$ and that there exist classes $a_1,\dots,a_k \in H^*(E;R)$ such that $a_i\vert_F = \alpha_i$. \red{Note that this is a strong assumption because there is some subtely about identification of $F$ with the fiber over a point in $B$.}
    
    Then the map $H^*(F;R)\otimes_R H^*(B;R) \to H^*(E;R)$ given by $x\otimes y \mapsto p^*y \cup a_i$ is an isomorphism of $R$-modules.
\end{theorem}
\begin{remark}
    This recovers Kunneth formula for the product $E = F\times B$ by taking $a_i = \alpha_i \otimes 1$. If $F$ has $R$-torsion, this cannot work because of Kunneth.
\end{remark}

\begin{example}
    [Nonexamples] $\R \to S^1$. $S^3 \subset \C^2 \to S^2 = \C\P^1$ the Hopf fibration. In the second case, we compute \begin{align*}
        H^*(\S^3; \Q) &= \Q \oplus \Q[-3] \\
        H^*(S^2; \Q) \otimes H^*(\S^1; \Q) &= \Q \oplus \Q[-1] \oplus \Q[-2] \oplus \Q[-3]
        \end{align*}
\end{example}

What is happening is that there is a grid \[\begin{bmatrix}
    \Z & 0 & \Z \\
    \Z & 0 & \Z \\
\end{bmatrix}\] and there is a differential $d_2$ from the $\Z$ in the bottom right to the $\Z$ in the top left which kills the $\Z$ in the bottom right and the $\Z$ in the top left. 

\begin{proof}

    \end{align*}
    Tensor with $H^*(F)$. Also have a sequence for $\pi^{-1}(B_n)$. Get a map from $H^*(F)\otimes H^*(B_n)$ to $H^*(\pi^{-1}(B_n))$ and check that it is an iso at each step. The key iput is that we have a map between the original complexes.
\end{proof}

\begin{theorem}

\begin{definition}[Characteristic classes]
    For a real line bundle $L\to X$ classified by $k_L: X \to \R\P^\infty$, the Stiefel-Whitney class of $L$ is $w(L) = k_L^* w$. For a complex line bundle $L\to X$ classified by $k_L: X \to \C\P^\infty$, the Chern class of $L$ is $c(L) = k_L^* c$. 
\end{definition}

        c_1(L\otimes L') &= c_1(L) + c_1(L')
    \end{align*}
\end{theorem}

\subsection{Homotopy groups}
\begin{definition}
    The $n$-th homotopy group of a pointed space $(X,x)$ is the set of homotopy classes of maps $S^n \to X$ sending the base point to $x$. We denote it by $\pi_n(X,x)$ or simply $\pi_n(X)$ if $X$ is path connected.
\end{definition}
The group law is given by writing $S^n \to S^n \vee S^n$ as the pinch map and then composing with the wedge of two maps $S^n \to X$. $\pi_n(X,x)$ can also be modeled by the set of homotopy classes of maps $I^n \to X$ sending the boundary to $x$. In this model, the group law can be defined by concatenating along the $1$st coordinate. 
The homotopy groups are abelian for $n\geq 2$.

There is an action of $\pi_1(X)$ on $\pi_n(X)$ for $n\geq 2$ by dragging the base point around loops. We say that $X$ is \textbf{$n$-simple} if the action of $\pi_1(X)$ on $\pi_n(X)$ is trivial, and $X$ is \textbf{simple} if it is $n$-simple for all $n$. 

\begin{proposition}
    A covering space projection $p: \tilde X \to X$ induces an isomorphism on $\pi_n$ for $n\geq 2$.
\end{proposition}

\begin{proof}
    The map is surjective because we can lift maps $S^n \to X$ to $\tilde X$ because $S^n$ is simply connected. The map is injective because if a map $S^n \to \tilde X$ becomes nullhomotopic in $X$, then we can lift the nullhomotopy to $\tilde X$ and get a nullhomotopy there.
\end{proof}

Moreover, if $\tilde X$ is the universal cover of $X$, then $\pi_1(X)$ acts on $\tilde X$ by deck transformations and this action induces the action of $\pi_1(X)$ on $\pi_n(\tilde X) \cong \pi_n(X)$. 

\begin{proposition}
    A product of path connected spaces $X_\alpha$ has homotopy groups given by the product of the homotopy groups of the factors, i.e. $\pi_n(\prod X_\alpha) \cong \prod \pi_n(X_\alpha)$.
\end{proposition}

\begin{proof}
This follows immediately from the universal property of products. 
\end{proof}

Another useful notion is the relative homotopy groups $\pi_n(X,A,x)$ for a pair of spaces $A \subset X$. \begin{definition}
Let $x\in A \subset X$ be a pointed pair of spaces. The $n$-th relative homotopy group $\pi_n(X,A,x)$ is the set
    of homotopy classes of maps $(I^n, \partial I^n,J^{n-1}) \to (X,A,x)$ where \[J^{n-1}=\overline{\partial I^n\setminus (I^{n-1}\times\{1\})}\] with homotopies through such maps. Equivalently, it is homotopy classes of maps $(D^n, S^{n-1}, *) \to (X,A,x)$ modulo homotopies through such maps. 
\end{definition}

    \end{align*} coming from the inclusion of pairs. The boundary map is the restriction of a map $D^n \to X$ to the boundary $S^{n-1} \to A$.
\end{theorem}

\subsection{Whitehead's theorem}
\begin{theorem}
    A map $f: X \to Y$ between CW complexes is a homotopy equivalence if and only if it induces isomorphisms on all homotopy groups.
\end{theorem}

Note that two CW complexes may have the same homotopy groups but not be homotopy equivalent. In particular, having a map inducing isomorphisms on homotopy groups is a much stronger condition than just having isomorphic homotopy groups. One situation where the latter implies the former is when all homotopy groups vanish, in which case the inclusion of a point is a homotopy equivalence.

\begin{example}
    Consider \(X=\mathbb{R}P^2\) and \(Y=S^2\times \mathbb{R}P^\infty\). These both have fundamental group \(\mathbb{Z}/2\Z\). Their higher homotopy groups are isomorphic since their universal covers \(S^2\) and \(S^2\times S^\infty\) are homotopy equivalent, \(S^\infty\) being contractible. But \(\mathbb{R}P^2\) and \(S^2\times \mathbb{R}P^\infty\) are not homotopy equivalent since their homology groups are vastly different, \(S^2\times \mathbb{R}P^\infty\) having nonvanishing homology in infinitely many dimensions since it retracts onto \(\mathbb{R}P^\infty\).
\end{example}

\begin{example}
Another pair of CW complexes that are not homotopy equivalent but have isomorphic homotopy groups is \(S^2\) and \(S^3\times \mathbb{C}P^\infty\). Indeed, \(\mathbb{C}P^\infty\) is an Eilenberg--MacLane space \(K(\mathbb{Z},2)\), so
\[
\pi_n(\mathbb{C}P^\infty)=
\begin{cases}
\mathbb{Z}, & n=2,\\
0, & n\neq 2.
\end{cases}
\]
Therefore
\[
\pi_n(S^3\times \mathbb{C}P^\infty)
\cong
\pi_n(S^3)\oplus \pi_n(\mathbb{C}P^\infty)
\]
for all \(n\ge 1\), and hence
\[
\pi_n(S^3\times \mathbb{C}P^\infty)\cong
\begin{cases}
\mathbb{Z}, & n=2,\\
\pi_n(S^3), & n\ge 3.
\end{cases}
\]

Now consider the Hopf fibration
\[
S^1 \longrightarrow S^3 \longrightarrow S^2.
\]
Its long exact sequence in homotopy implies that
\[
\pi_n(S^3)\cong \pi_n(S^2)
\qquad\text{for all } n\ge 3,
\]
since \(\pi_n(S^1)=0\) for \(n\ge 2\). Also,
\[
\pi_2(S^2)\cong \mathbb{Z},
\qquad
\pi_1(S^2)=0.
\]
Since \(S^3\times \mathbb{C}P^\infty\) is simply connected as well, it follows that
\[
\pi_n(S^3\times \mathbb{C}P^\infty)\cong \pi_n(S^2)
\qquad\text{for all } n\ge 1.
\]

However, these spaces are not homotopy equivalent. For example,
\[
H_4(S^2)=0,
\]
while
\[
H_4(S^3\times \mathbb{C}P^\infty)
\cong
H_4(\mathbb{C}P^\infty)
\cong
\mathbb{Z},
\]
so their homology groups are not isomorphic. Therefore \(S^2\) and \(S^3\times \mathbb{C}P^\infty\) cannot be homotopy equivalent.
\end{example}

\subsection{Cellular approximation and CW approximation}
A map $f: X \to Y$ is \textbf{cellular} if it sends the $n$-skeleton of $X$ to the $n$-skeleton of $Y$ for all $n$. \begin{theorem}
    Any map $f:X\to Y$ between CW complexes is homotopic to a cellular map. If $f$ is already cellular on a subcomplex $A\subset X$, then the homotopy can be taken to be relative to $A$.
\end{theorem}
It immediately follows that $\pi_k(S^n) = 0$ for $k < n$ because the $k$-skeleton of $S^n$ is a point.

\begin{definition}
    A map $f: X \to Y$ is a weak homotopy equivalence if it induces isomorphisms on all homotopy groups. 
\end{definition}
Whitehead's theorem reformaulated says that a weak homotopy equivalence between CW complexes is a homotopy equivalence. However, a weak homotopy equivalence between general spaces need not be a homotopy equivalence. For example, there exist spaces which have trivial homotopy groups but are not contractible. For these spaces, the constant map to a point is a weak homotopy equivalence but not a homotopy equivalence.

\begin{example}
    Consider the quasi-circle $Y$ which is a closed subspace of $\R^2$ obtained by taking a portion of the graph of $y = \sin(1/x)$, the segment $[-1,1]$ in the $y$-axis, and connecting them with a curve. This space comes with a map $p: Y \to S^1$ which collapses the sine curve and the interval to a point. Then this space has trivial homotopy groups. Any map from $S^n$ to this space is nullhomotopic because the image is compact, so it must meet the sine-curve in finitely many oscillations. This will clearly be nullhomotopic.  However, the quasi-circle is not contractible because if it were, then $p$ would be nullhomotopic and we could lift $p$ to the universal cover $\R$ of $S^1$. The lifted map $\tilde p: Y \to \R$ in fact factors through $Y/I = S^1$ because the quotient map collapses $I$ to a point. Then this gives a section of the universal cover $\R \to S^1$, which is impossible.
\end{example}

\begin{definition}
    Let $Y$ be a topological space. A \textit{CW--approximation} of $Y$ is a CW complex $X$ together with a map $f:X\to Y$ such that for every CW complex $Z$, the induced map $f_*:[Z,X]\to [Z,Y]$ is a bijection, where $[-,-]$ denotes the set of homotopy classes of maps.
\end{definition}
Taking $Z = S^n$ for all $n$ shows that a CW approximation is a weak homotopy equivalence. Conversely, if $f:X\to Y$ is a weak homotopy equivalence and $X$ is a CW complex, then $f$ is a CW approximation by Whitehead's theorem. 

\begin{theorem}
    Every topological space $Y$ admits a CW approximation $X\to Y$. If $X$ is pathconnected, $X$ can be chosen to have a single 0 cell, with all other cells attached by basepoint-preserving maps. 
\end{theorem}

There is also a relative version of CW approximation.  

\begin{theorem}[Relative CW approximation]
Let $(X,A)$ be a pair of spaces with $A$ a CW complex (or more generally a
subspace). Then there exists a relative CW complex $(Z,A)$ and a map of pairs
\[
f:(Z,A)\longrightarrow (X,A)
\]
such that:
\begin{enumerate}
    \item $f|_A = \mathrm{id}_A$,
    \item $f$ is a weak homotopy equivalence of pairs, i.e.
    \[
    \pi_k(Z,A) \xrightarrow{\;\cong\;} \pi_k(X,A)
    \quad \text{for all } k.
    \]
\end{enumerate}
If $(X,A)$ is $n$-connected,
i.e.
\[
\pi_k(X,A)=0 \quad \text{for all } k\le n,
\]
then there exists a relative CW complex $(Z,A)$ and a weak equivalence of
pairs
\[
(Z,A)\xrightarrow{\;\simeq\;} (X,A)
\]
such that $(Z,A)$ is obtained from $A$ by attaching no cells in dimensions $\le n$. Equivalently, the first relative cells occur in dimension $n+1$.
\end{theorem}

\subsection{Homotopy excision}

Let $X$ and $Y$ be subcomplexes of a CW complex $Z=X\cup Y$, and let
\[
A=X\cap Y.
\]
Assume that the pairs $(X,A)$ and $(Y,A)$ are $m$-connected and
$n$-connected respectively. Then inclusion of pairs induces a map
\[
\pi_k(X,A)\longrightarrow \pi_k(X\cup Y,\,Y).
\]

\begin{theorem}[Homotopy excision]
The map
\[
\pi_k(X,A)\longrightarrow \pi_k(X\cup Y,\,Y)
\]
is an isomorphism for $k<m+n$ and a surjection for $k=m+n$.
\end{theorem}


\begin{remark}
A useful special case is when $Y=CA$ is the cone on $A$. Then
\[
X\cup_A CA \simeq X/A,
\]
and since $CA$ is contractible, we get
\[
\pi_k(X\cup_A CA,\,CA)\cong \pi_k(X/A).
\]
Thus homotopy excision gives a comparison map
\[
\pi_k(X,A)\longrightarrow \pi_k(X/A)
\]
which is an isomorphism for $k<m+n$ and a surjection for $k=m+n$.
\end{remark}

\begin{example}
Suppose
\[
X=A\cup_f e^{m+1},
\qquad
Y=A\cup_g e^{n+1}.
\]
Then $(X,A)$ is $m$-connected and $(Y,A)$ is $n$-connected. In this case the theorem implies that
\[
\pi_k(X,A)\oplus \pi_k(Y,A)\longrightarrow \pi_k(X\cup Y,\,A)
\]
is an isomorphism for $k<m+n$ and a surjection for $k=m+n$.
\end{example}


In low degrees the relative homotopy of $X\cup Y$ is just the sum of the
two pieces. Beyond this range, interaction terms can appear; the first such
new classes are related to Whitehead products.
\begin{theorem}[Freudenthal suspension theorem]
Let $X$ be an $(n-1)$-connected CW complex, with $n\ge 1$. Then the suspension map
\[
\Sigma \colon \pi_k(X)\longrightarrow \pi_{k+1}(\Sigma X)
\]
is an isomorphism for $k<2n-1$ and a surjection for $k=2n-1$.
\end{theorem}

\begin{example}
Take $X=S^2$. Since $S^2$ is $1$-connected, we may take $n=2$. The theorem then says that
\[
\pi_2(S^2)\xrightarrow{\;\cong\;} \pi_3(S^3),
\]
which is indeed the isomorphism
\[
\mathbb Z \xrightarrow{\;\cong\;} \mathbb Z.
\]
It also says that the suspension map
\[
\pi_3(S^2)\twoheadrightarrow \pi_4(S^3)
\]
is surjective. In fact,
\[
\pi_3(S^2)\cong \mathbb Z,
\qquad
\pi_4(S^3)\cong \mathbb Z/2.
\]
\end{example}

\begin{proof}[Idea of proof]
Apply the homotopy excision theorem to the decomposition
\[
\Sigma X = CX_+ \cup_X CX_-,
\]
where $CX_+$ and $CX_-$ are two copies of the cone on $X$, glued along their common base $X$.

Since each cone is contractible, the pair $(CX_\pm,X)$ is $n$-connected when $X$ is $(n-1)$-connected. Thus Blakers--Massey gives that the natural map
\[
\pi_k(CX_+,X)\longrightarrow \pi_k(\Sigma X,CX_-)
\]
is an isomorphism for $k<2n$ and a surjection for $k=2n$.

Now use the identifications
\[
\pi_k(CX_+,X)\cong \pi_{k-1}(X),
\qquad
\pi_k(\Sigma X,CX_-)\cong \pi_k(\Sigma X),
\]
the second because $CX_-$ is contractible. Under these identifications, the map above becomes exactly the suspension map
\[
\pi_{k-1}(X)\longrightarrow \pi_k(\Sigma X).
\]
\end{proof}

\begin{example}[Whitehead products]
Let us compute
$\pi_3\!\left(\bigvee_\alpha S^2_\alpha\right)$,
showing that it is free abelian with basis consisting of the Hopf maps
$S^3 \to S^2_\alpha \subset \bigvee_\alpha S^2_\alpha$

$S^2_\alpha \times S^2_\beta$, for all unordered pairs $\alpha \neq \beta$.

Suppose first that there are only finitely many summands $S^2_\alpha$. For a finite product
$\prod_\alpha X_\alpha$
of path-connected spaces, the map
$\pi_n\!\left(\bigvee_\alpha X_\alpha\right)\longrightarrow \pi_n\!\left(\prod_\alpha X_\alpha\right)$
induced by inclusion is surjective, since
$\pi_n\!\left(\prod_\alpha X_\alpha\right)\cong \bigoplus_\alpha \pi_n(X_\alpha)$
is generated by the subgroups $\pi_n(X_\alpha)$. Thus the long exact sequence of homotopy groups for the pair
$\left(\prod_\alpha X_\alpha,\ \bigvee_\alpha X_\alpha\right)$
breaks up into short exact sequences
\[
0 \longrightarrow \pi_{n+1}\!\left(\prod_\alpha X_\alpha,\ \bigvee_\alpha X_\alpha\right)
\longrightarrow \pi_n\!\left(\bigvee_\alpha X_\alpha\right)
\longrightarrow \pi_n\!\left(\prod_\alpha X_\alpha\right)
\longrightarrow 0.
\]

These short exact sequences split, since the inclusions
$X_\alpha \hookrightarrow \bigvee_\alpha X_\alpha$
induce maps
$\pi_n(X_\alpha)\to \pi_n\!\left(\bigvee_\alpha X_\alpha\right)$,
and hence a splitting homomorphism
$\bigoplus_\alpha \pi_n(X_\alpha)\to \pi_n\!\left(\bigvee_\alpha X_\alpha\right)$.

Taking $X_\alpha=S^2_\alpha$ and $n=3$, we get an isomorphism
\[
\pi_3\!\left(\bigvee_\alpha S^2_\alpha\right)
\cong
\pi_4\!\left(\prod_\alpha S^2_\alpha,\ \bigvee_\alpha S^2_\alpha\right)
\oplus
\left(\bigoplus_\alpha \pi_3(S^2_\alpha)\right).
\]

The factor
$\bigoplus_\alpha \pi_3(S^2_\alpha)$
is free with basis the Hopf maps $S^3\to S^2_\alpha$. For the other factor, we have
\[
\pi_4\!\left(\prod_\alpha S^2_\alpha,\ \bigvee_\alpha S^2_\alpha\right)
\cong
\pi_4\!\left(\prod_\alpha S^2_\alpha \Big/ \bigvee_\alpha S^2_\alpha\right).
\]
The quotient
$\prod_\alpha S^2_\alpha \Big/ \bigvee_\alpha S^2_\alpha$
has $5$-skeleton a wedge of spheres
$\bigvee_{\alpha\beta} S^4_{\alpha\beta}$
for $\alpha\neq\beta$, so
\[
\pi_4\!\left(\prod_\alpha S^2_\alpha \Big/ \bigvee_\alpha S^2_\alpha\right)
\cong
\pi_4\!\left(\bigvee_{\alpha\beta} S^4_{\alpha\beta}\right),
\]
which is free with basis the inclusions
$S^4_{\alpha\beta}\hookrightarrow \bigvee_{\alpha\beta} S^4_{\alpha\beta}$.
Hence
$\pi_4\!\left(\prod_\alpha S^2_\alpha,\ \bigvee_\alpha S^2_\alpha\right)$
is free with basis the characteristic maps of the $4$-cells
$e^2_\alpha\times e^2_\beta$.
Via the injection
\[
\partial:\pi_4\!\left(\prod_\alpha S^2_\alpha,\ \bigvee_\alpha S^2_\alpha\right)
\to
\pi_3\!\left(\bigvee_\alpha S^2_\alpha\right),
\]
this means that the attaching maps of the cells
$e^2_\alpha\times e^2_\beta$
form a basis for the summand
$\operatorname{Im}\partial \subset \pi_3\!\left(\bigvee_\alpha S^2_\alpha\right)$.
This finishes the proof for the case of finitely many summands $S^2_\alpha$. The case of infinitely many summands follows immediately since any map
$S^3\to \bigvee_\alpha S^2_\alpha$

\end{align*}
which gives a functorial CW approximation. Moreover, $\operatorname{Sing}$ and $|{-}|$ become inverse equivalences on the homotopy category.

\subsection{Group-like monoids and the bar construction}
\textbf{Recall the slogan from last time: in homotopy theory, group-like monoids are as good as groups.}
An $A_\infty$-space is the homotopy-theoretic replacement for a strictly associative topological monoid, and one can build its bar construction via the geometric realization of the simplicial bar construction.

There is an algebraic analog. Suppose $A$ is an associative algebra and $M,N$ are right and left $A$-modules. If one wants to compute the tensor product correctly, in the derived sense, one resolves and forms a bar complex. A universal way to think about this is:
\begin{align*}
    M \otimes_A N
    \simeq (M \otimes_k N) \otimes_{A \otimes A^{\mathrm{op}}} A,
\end{align*}
so one resolves $A$ functorially as an $A \otimes A^{\mathrm{op}}$-module. This leads to the bar resolution
\begin{align*}
    A \longleftarrow A^{\otimes 2} \longleftarrow A^{\otimes 3} \longleftarrow A^{\otimes 4} \longleftarrow \cdots,
\end{align*}
where the face maps are given by multiplying adjacent factors, and the differential is the alternating sum of the face maps.

For example, if $M=N=k$ with augmentation $A \to k$, one gets the complex
\begin{align*}
    k \longleftarrow A \longleftarrow A^{\otimes 2} \longleftarrow \cdots,
\end{align*}
whose homology computes
\begin{align*}
    \Tor_*^A(k,k) \cong k \otimes_A^{\mathbf{L}} k.
\end{align*}
Any two such resolutions are quasi-isomorphic.

Topologically, if $M$ is a monoid, one considers
\begin{align*}
    * \longleftarrow M \longleftarrow M^2 \longleftarrow M^3 \longleftarrow \cdots,
\end{align*}
and fills in the simplices by homotopies. If $M$ acts trivially on a point, replacing all copies of $M$ by $*$ gives $BM$; replacing only one component gives $EM$.

More generally, if $M$ acts on the right on $X$ and on the left on $Y$, the \textbf{homotopy quotient / two-sided bar construction} is
\begin{align*}
    X \times_M^h Y.
\end{align*}
modeled by the simplicial space
\begin{align*}
    X \times Y \Longleftarrow X \times M \times Y \Longleftarrow X \times M^2 \times Y \Longleftarrow \cdots.
\end{align*}
A homotopy-associative action is the same as filling in the simplices in this bar construction.

There is also the stricter simplicial replacement: a simplicial diagram
\begin{align*}
    * \Longleftarrow M \Longleftarrow M^2 \Longleftarrow M^3 \Longleftarrow \cdots
\end{align*}
whose vertical maps to the corresponding powers are homotopy equivalences.
Thus:
\begin{align*}
    \{\text{topological groups}\}
    &\longleftrightarrow \{A_\infty\text{-group-like monoids}\}, \\
    \{\text{topological actions}\}
    &\longleftrightarrow \{A_\infty\text{-actions}\}.
\end{align*}
Likewise, ordinary quotients are replaced by homotopy quotients:
\begin{align*}
    (X \times Y)/G \qquad \leadsto \qquad X \times_G^h Y \simeq \operatorname{Bar}(X,G,Y).
\end{align*}
If the action is free and $X \times Y \to (X \times Y)/G$ is a fiber bundle, then the comparison map from the homotopy quotient to the ordinary quotient is a homotopy equivalence.

\subsection{Principal fibrations and homotopy fibers}
By this definition, any fibration
\begin{align*}
    F \hookrightarrow E \to B
\end{align*}
is associated to a principal homotopy-group fibration.
For a path-connected base, one has the path-loop fibration
\begin{align*}
    \Omega B \hookrightarrow P_bB \to B
\end{align*}
and the loop space $\Omega B$ acts on the fiber $F_b$.

A principal fibration is a fibration equipped with a translation action by a loop space. In particular, since principal bundles are classified, one has the usual picture
\begin{center}
\begin{tikzcd}
E \arrow[r] \arrow[d] & EG \arrow[d] \\
B \arrow[r] & BG
\end{tikzcd}
\end{center}
for a principal $G$-bundle.

If $B \to X$ is a map and $b \in B$ maps to the base point $x \in X$, then one gets a strict homomorphism
\begin{align*}
    \Omega_b B \to \Omega_x X,
\end{align*}
and hence an action of $\Omega X$ on the homotopy fiber.
The total space $E$ is the homotopy fiber of the classifying map $B \to X$.

Consequently, for any fibration $F \to E \to B$ one obtains a Puppe sequence
\begin{align*}
    \Omega E \to \Omega B \to F \to E \to B,
\end{align*}
and by continuing to the left and right one gets a sequence of spaces in which the composite of any two consecutive maps is null-homotopic, with a specified null-homotopy, and any three consecutive terms exhibit the leftmost one as the homotopy fiber of the right-hand map.
Taking $\pi_0$ gives the long exact sequence in homotopy. Moreover, the original fibration is principal if and only if this sequence can be continued one more step to the right.

\subsection{Eilenberg--MacLane spaces}
Let $\pi$ be a group, with $\pi$ abelian if $n>1$. A space $K(\pi,n)$ is a connected space with
\begin{align*}
    \pi_i(K(\pi,n)) =
    \begin{cases}
        \pi & i=n, \\
        0 & i\neq n.
    \end{cases}
\end{align*}
It is unique up to weak homotopy equivalence.

For $n \ge 2$, the Hurewicz theorem gives
\begin{align*}
    H_n(K(\pi,n);\Z) \cong \pi,
    \qquad
    H_i(K(\pi,n);\Z)=0 \quad (i<n).
\end{align*}
Hence the universal class in degree $n$ gives
\begin{align*}
    H^n(K(\pi,n);\pi) \cong \Hom(\pi,\pi),
\end{align*}
and in particular a distinguished class corresponding to the identity map.

The main representability statement is
\begin{align*}
    [X,K(\pi,n)] \cong H^n(X;\pi).
\end{align*}
For $n=1$ and $\pi$ abelian this agrees with the classification of principal $\pi$-bundles:
\begin{align*}
    [X,B\pi]
    = \{\text{isomorphism classes of principal } \pi\text{-bundles on }X\}
    = H^1(X;\pi)
\end{align*}


The Yoneda lemma implies that $K(\pi,n)$ is unique up to isomorphism in the homotopy category.
A sketch with CW complexes is as follows: if $K(\pi,n)$ has no cells below dimension $n$, then any map $X \to K(\pi,n)$ factors through $X/X_{n-1}$, and one gets a homomorphism
\begin{align*}
    C_n(X) \to \pi
\end{align*}
which vanishes on boundaries exactly when the map extends.

Also note that if $\pi$ is abelian, then $B\pi$ is an abelian topological group up to homotopy, and so is $\Omega B\pi \simeq \pi$ in the appropriate range.

\subsection{Principal fibration theorem and Postnikov towers}
\begin{theorem}[Principal fibration theorem]
Let $B$ be simply connected, and let
\[
K(\pi,n)\simeq F \hookrightarrow E \xrightarrow{p} B
\]
be a fibration with $n>1$. Then the monodromy action of $\pi_1(B)$ on
$\pi_n(F)\cong \pi$ is trivial, and $p$ is a principal $K(\pi,n)$-fibration
up to fiber homotopy equivalence. Consequently there is a classifying map
\[
c_E : B \to BK(\pi,n)\simeq K(\pi,n+1)
\]
such that $E$ is the homotopy fiber of $c_E$. Hence fiber homotopy equivalence
classes of such fibrations are in bijection with
\[
[B,BK(\pi,n)] \cong [B,K(\pi,n+1)] \cong H^{n+1}(B;\pi).
\]
The corresponding class
\[
k(E)\in H^{n+1}(B;\pi)
\]
is called the $n$th Postnikov-invariant of the fibration.
\end{theorem}

If $B$ is not simply connected, then the monodromy action $\rho$ of $\pi_1(B)$ on $\pi_n(F) = \pi$ may be nontrivial. In this case, one has the $n$th Postnikov invariant $k(E) \in H^{n+1}(B;\pi_\rho)$, where $\pi_\rho$ is the local system on $B$ associated to the representation $\rho$.

\begin{remark}
    Suppose $X$ is a connected CW complex and $\mathcal L$ corresponds to a $\pi_1(X)$-module $A$ is a local coefficient system. Then the universal cover $\widetilde X$ has a cellular chain complex $C_*(\widetilde X)$
and this is naturally a module over $\mathbb Z[\pi_1(X)]$, because deck transformations act on $\widetilde X$.

Then homology with local coefficients is
\[H_*(X;\mathcal L)=H_*\big(C_*(\widetilde X)\otimes_{\mathbb Z[\pi_1(X)]} A\big)\] and cohomology with local coefficients is
\[
H^*(X;\mathcal L)=H^*\big(\Hom_{\mathbb Z[\pi_1(X)]}(C_*(\widetilde X),A)\big).
\]

\end{remark}

\begin{theorem}[Postnikov tower]
For a simply connected space $X$ with homotopy groups $\pi_2,\pi_3,\dots$, there are spaces $X_{\le n}$ and maps
\begin{align*}
    X \to X_{\le n}
\end{align*}
such that
\begin{enumerate}
    \item the induced map $\pi_k(X) \to \pi_k(X_{\le n})$
    is an isomorphism for $k \le n$;
    \item $\pi_k(X_{\le n}) = 0$ for $k > n$;
    \item there are fibrations
    \begin{align*}
        K(\pi_{n+1}(X),n+1) \to X_{\le n+1} \to X_{\le n},
    \end{align*}
    classified by classes $k_{n+1} \in H^{n+2}(X_{\le n};\pi_{n+1}(X))$
    the $(n+1)$st Postnikov $k$-invariants.
    \item The natural map $X \to \varprojlim_n X_{\le n}$
is a weak homotopy equivalence.
\end{enumerate}
\end{theorem}



The Postnikov tower is thus built from Eilenberg--MacLane fibers step by step, and is dual to building a space by attaching cells or, homologically, by Moore spaces.

\subsection{Dold--Thom and Moore spaces}
The Dold--Thom theorem says that the infinite symmetric product detects homology. For a based space $X$, define
\begin{align*}
    \operatorname{SP}^\infty(X)
    = \varinjlim_m X^m/S_m.
\end{align*} $\operatorname{SP}^\infty(X)$ is a connected topological abelian group
\begin{theorem}[Dold--Thom]
For a based connected CW complex $X$,
\[\pi_n(\operatorname{SP}^\infty(X)) \cong \widetilde H_n(X;\mathbb Z)
\qquad (n\ge 1)\]
Since $\operatorname{SP}^\infty(X)$ is a connected topological abelian group, all Postnikov invariants vanish, and we have a homotopy equivalence \[
\operatorname{SP}^\infty(X)\simeq \prod_{n\ge 1} K(\widetilde H_n(X;\mathbb Z),n)
\]
\end{theorem}
In particular, topological abelian groups decompose as products of Eilenberg--MacLane spaces. 

\begin{example}
A point of $\operatorname{SP}^m(\mathbb CP^1)$ is an unordered m-tuple of points on $\mathbb CP^1$, i.e. an effective divisor of degree $m$ on $\mathbb CP^1$. Such a divisor corresponds to a homogeneous polynomial of degree m up to scale, namely the polynomial whose zeros are those points with multiplicity. So
$\operatorname{SP}^m(\mathbb CP^1)\cong \mathbb CP^m$.
Passing to the colimit over $m$ gives
\begin{align*}
    \C P^\infty \simeq \operatorname{SP}^\infty(\C P^1).
\end{align*}
In particular, the group structure on $BS^1 \simeq
\C P^\infty$ which is only defined up to homotopy can be made into a strict group structure by identifying it with $\operatorname{SP}^\infty(\C P^1)$.
\end{example}

Dually, a Moore space $M(\pi,n)$ is a space with one prescribed reduced homology group:
\begin{align*}
    \widetilde H_n(M(\pi,n)) \cong \pi,
    \qquad
    \widetilde H_i(M(\pi,n)) = 0 \text{ for } i\neq n.
\end{align*}
The Postnikov tower is in this sense dual to a cell decomposition / Moore-space decomposition of a space.

\subsection{Construction of Postnikov tower}
Let $X$ be simply connected. Then there is a tower of fibrations $X_i \to X_{i-1}$ with fiber $K(\pi_i(X),i)$, and $X$ maps to all of them.

\begin{theorem}
    This exists for every simply connected space $X$. It is homotopy functorial in the sense that a map $X \to Y$ induces a map between the towers. Moreover, $\pi_k(X)\to \pi_k(X_i)$ is an isomorphism for $k \le i$. The map $X \to \varprojlim X_i$ is a weak homotopy equivalence because the fiber of $X_{\leq m} \to X_{\leq n}$ are $n$-connected, so no lifting problem for $\leq n$-spheres.
\end{theorem}

The construction of $X_{\leq n}$ is by killing homotopy groups above $n$ by attaching $n+2$-cells and higher. We claim that attaching an $n+2$-cell to $X$ by $\phi:S^{n+1}\to X$ kills the class of $[\phi]$ in $\pi_{n+1}(X)$ and does not change $\pi_{\leq n}(X)$. It does change things higher up, but then you can inductively kill.

This claim follows from the excision theorem. If $X' = X \cup_\phi e^{n+2}$ and $X$ is connected, then the long exact sequence of the pair $(X',X)$ gives 
\begin{align*}
    \dots \to \pi_{k+1}(X) \to \pi_{k+1}(X') \to \pi_{k+1}(X',X) \to \pi_k(X) \to \pi_k(X') \to \dots
\end{align*} By the excision theorem \begin{align*}
    \pi_k(X',X) \cong \pi_k(X'/X) \cong \pi_k(S^{n+2})
\end{align*} for $k \leq n+1$. Note that $X$ connected implies that $(X',X)$ is $n+1$-connected. Hence $\pi_k(X) \to \pi_k(X')$ is an isomorphism for $k \leq n$ and the map $\pi_{n+2}(X',X) \to \pi_{n+2}(S^{n+2})$ is surjective, i.e. the connecting map kills $[\phi] \in \pi_{n+1}(X)$.

\subsection{Obstruction theorem}
Suppose that we have a strictly commuting square of a cofibration $A \hookrightarrow X$ and a fibration $E \to B$:
\begin{center}
\begin{tikzcd}
A \arrow[r] \arrow[d,hook] & E \arrow[d] \\
X \arrow[r] & B
\end{tikzcd}
\end{center}
The question is does there exist a lift $X \to E$ making the diagram commute? Note that if we omit the cofibration/fibration condition, then we can only ask for commutation up to homotopy. If $B = *$ we are asking for an extension of a map $A \to E$ to $X$. If $A = *$, we are asking for a lift of a map $X \to B$ to $E$.  

\begin{theorem}
    Assume that one of the following holds.
    \begin{enumerate}
        \item $X$ is obtained from $A$ by sttaching only $n+1$ cells.
        \item $E$ is a principal $K(\pi,n)$-fibration over $B$ simply connected.
    \end{enumerate} Then the square defines an obstruction class $\omega^{n+1} \in H^{n+1}(X,A;\pi_n(F))$ whose vanishing is necessary and sufficient for the existence of a lift $X \to E$ making the diagram commute. 
    
    Moreover, if $\omega^{n+1} = 0$ and $X$ is obtained from $A$ by attaching only $n+1$ cells, then the set of lifts up to homotopy form a torsor under $H^{n+1}(X,A;\pi_{n+1}(F))$. If $\omega^{n+1} = 0$ and $E$ is a principal $K(\pi,n)$-fibration, then the set of lifts up to homotopy form a torsor under $H^n(X,A;\pi)$.
\end{theorem}



Remark that this criterion can be used iteratively. Check for a relative cell decomposition of $X$ relative to $A$ and go one dimension at a time, or check for the relative Postnikov tower of $E$ relative to $B$ and go one Postnikov stage at a time.

\begin{corollary}
    If either $A \subset X$ or $E\to B$ is a weak homotopy equivalence, then the lift exists and is unique up to homotopy.
\end{corollary}
This is clear for fibrations since $\pi_* F = 0$. For cofibrations, while $H^*(X,A;\pi_n(F)) = 0$ an extra check is needed because we have to go one step at a time.

\begin{example}
Consider the path-space fibration
\[
K(\pi,n)\simeq \Omega K(\pi,n+1)\longrightarrow PK(\pi,n+1)\longrightarrow K(\pi,n+1),
\]
whose total space is contractible. Up to homotopy this is the universal principal \(K(\pi,n)\)-fibration
\[
K(\pi,n)\to *\to K(\pi,n+1).
\]
Given a map \(f:X\to K(\pi,n+1)\), a lift to \(*\) exists if and only if \(f\) is nullhomotopic. Under the natural bijection
\[
[X,K(\pi,n+1)]\cong H^{n+1}(X;\pi),
\]
the homotopy class of \(f\) corresponds to the pullback \(f^*(u)\) of the universal class
\[
u\in H^{n+1}(K(\pi,n+1);\pi).
\]
Hence the obstruction to lifting \(f\) is precisely \(f^*(u)\).\end{example}

Application is the functoriality of Postnikov towers. Assume $X$ and $Y$ are simply connected and $f:X \to Y$ is a map. We want to show that there is a map $X_{\leq n} \to Y_{\leq n}$. The point is that $X_{\leq n}$ is obtained from $X$ by attaching cells of dimension $n+2$ and higher, and the attaching maps are null-homotopic in $Y_{\leq n}$, so we can extend the map $X \to Y_{\leq n}$ to $X_{\leq n}$.

This statement is analagous to the cellular approximation theorem, which says that if $X$ is a CW complex and $Y$ is a space, then any map $X \to Y$ is homotopic to a cellular map. 

\begin{proof}
    [Proof of the obstruction theorem]  Consider the case of extending across cells. A lifting on each cell exists if and only if $[f \circ \phi_\alpha] \in \pi_n(E)$ is $0$ for each attaching map $\phi_\alpha: S^n \to X$, and $f$ is the map $A\to E$. We know that $[p \circ f \circ \phi_\alpha] = 0$ as classes in $\pi_n(B)$ which means that $[f \circ \phi_\alpha]$ came from $\pi_n(F)$. We have defined the class in $H^{n+1}(X,A;\pi_n(F))$. This class cannot be killed unless it was $0$ since there are no higher cells.

As for the case of the principal fibration, the fibration $E\to B$ is classified by a map $B \to K(\pi,n+1)$, and the composition $X \to B \to K(\pi,n+1)$ defines a class in $(X,A)$, and a nullhomotopy of the map to $K(\pi,n+1)$ corresponds precisely to a relative lifting to $E$.
\end{proof}

\subsection{Spectral sequences and examples}
The main example is the Leray spectral sequence. It computes the cohomology of the total space of a fibration $F \to E \to B$ from the cohomology of the base and the fiber. We will assume $B$ is simply connected. The starting idea is to use a cell decomposition of $B$ and over each cell, the fibration is fiber homotopy equivalent to a product.

Know how to build a complex by filtering by the skeleta $B_n$ and $E_n = p^{-1}(B_n)$. We can keep increasing $n$ by the long exact sequence \begin{align*}
    \cdots \to H^*(E_n,E_{n-1};\Z) \to H^*(E_n;\Z) \to H^*(E_{n-1};\Z) \to \cdots
\end{align*} and we actually know how to describe $H^*(E_n,E_{n-1};\Z)$. It is the product \begin{align*}
    \prod_{n+1 cells in B} H^{* - (n+1)}(F;\Z)
\end{align*}
This is not an efficient method because one runs into trouble inductively. It also fails to exploit the algebraic ring structure on $H^*(E;\Z)$ and the module structure of $H^*(E;\Z)$ over $H^*(B;\Z)$. The better thing to do is to carry out an order by order computation of cohomology of a filtered complex. The analogy is to use Taylor series to compute solutions of ODEs. Each step is computing $H^*$ of a complex.

\begin{theorem}
    [Spectral sequence of a filtered complex] Consider a cohomologically graded complex \begin{align*}
    \cdots \to C^{n-1} \to C^n \to C^{n+1} \to \cdots
    \end{align*} with a decreasing filtration
\begin{align*}
    \cdots \supset F^p C^* \supset F^{p+1} C^* \supset \cdots
\end{align*} satisfing the Hausdorff condition $\bigcap_p F^p C^* = 0$ and completeness condition $C^* \cong \bigcup_p F^p C^*$. Then there exist a sequence of pages $E_r^{p,q}$ bigraded abelian groups with differentials $d_r: E_r^{p,q} \to E_r^{p+r,q-r+1}$ so that $E_{r+1}^{p,q} \cong H^{p,q}(E_r,d_r)$ and $d_r$ are the successive terms of $\delta$ in $F$. Moreover, $E_1^{p,q} \cong H^{p+q}(F^p C^*/F^{p+1} C^*)$. 

We also have a short exact sequence \begin{align*}
    0\to \lim^1 \bigoplus_{p+q=n-1} E^{p,q}_\infty 
    \to \Gr^F H^*(C^\bullet) \to \bigoplus_{p+q = *} E_\infty^{p,q} \to 0
\end{align*}
The spectral sequence converges if $\lim^1 \bigoplus_{p+q=n-1} E^{p,q}_\infty = 0$. A sufficient condition for convergence is for each $\bullet$, $F_n C^\bullet = 0$ for $n$ sufficiently large. One really hopes that the $\lim^1$ term vanishes. If it doesn't you probably set the problem up wrong.
\end{theorem}

\begin{example}
